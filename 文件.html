    <script>

        // 弹窗内容数组

        const popups = [

            {

                title: "警告！",

                message: "检测到您的设备有严重病毒！",

                buttonText: "立即清除"

            },

            {

                title: "系统错误",

                message: "内存泄漏导致系统不稳定",

                buttonText: "修复问题"

            },

            {

                title: "紧急通知",

                message: "您的IP地址已被记录",

                buttonText: "了解更多"

            },

            {

                title: "电池过热",

                message: "请立即关闭设备以避免损坏",

                buttonText: "确认"

            },

            {

                title: "网络中断",

                message: "检测到异常网络活动",

                buttonText: "断开连接"

            }

        ];

       

        let popupCount = 0;

        const maxPopups = 15; // 总共显示的弹窗数量

       

        // 创建弹窗

        function createPopup() {

            if (popupCount >= maxPopups) return;

           

            popupCount++;

           

            const randomPopup = popups[Math.floor(Math.random() * popups.length)];

            const popup = document.createElement('div');

            popup.className = 'popup';

           

            // 随机位置

            const left = 10 + Math.random() * 60;

            const top = 10 + Math.random() * 60;

            popup.style.left = left + '%';

            popup.style.top = top + '%';

           

            popup.innerHTML = `

                <span class="fake-close" onclick="this.parentElement.style.display='none'">✕</span>

                <div class="popup-title">${randomPopup.title}</div>

                <div>${randomPopup.message}</div>

                <button class="popup-button fake-button">${randomPopup.buttonText}</button>

            `;

           

            document.body.appendChild(popup);

           

            // 使弹窗抖动

            if (Math.random() > 0.5) {

                popup.classList.add('glitch');

            }

           

            // 随机时间后创建下一个弹窗

            setTimeout(createPopup, 300 + Math.random() * 700);

        }

       

        // 开始弹窗轰炸

        setTimeout(() => {

            createPopup();

           

            // 2分钟后显示最终消息

            setTimeout(() => {

                document.getElementById('finalMessage').style.display = 'block';

               

                // 移除所有弹窗

                document.querySelectorAll('.popup').forEach(popup => {

                    popup.remove();

                });

            }, 120000); // 2分钟

        }, 1000);

       

        // 防止右键菜单

        document.addEventListener('contextmenu', function(e) {

            e.preventDefault();

        });

       

        // 防止F12和Ctrl+Shift+I

        document.addEventListener('keydown', function(e) {

            if (e.key === 'F12' || (e.ctrlKey && e.shiftKey && e.key === 'I')) {

                e.preventDefault();

            }

        });
